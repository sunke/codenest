Products = CLASS {
     PROPERTIES {
         end stock = start stock + purchase - planned sales;
         end stock unit cost = 0;
         external demand = UNDEFINED;
         external demand >= planned sales;
         holding cost = end stock * end stock unit cost;
         planned sales = UNDEFINED;
         price = 0;
         purchase = UNDEFINED;
         purchase cost = purchase * purchase unit cost;
         purchase unit cost = 0;
         revenue = planned sales * price;
         start stock = UNDEFINED;
     };
     graphinfo = "{MgiHeight = 20; MgiState = 32; MgiWidth = 159; MgiXpos = 497; MgiYpos = 260; }";
     icon = "Manufacturing/beer_bottle";
     resource name = "Product";
     sort code = 100;
};
